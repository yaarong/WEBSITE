1.1 האינטרפייס ICommand.h

ב-C++ אין interface מובנה. כותבים מחלקה רגילה שכל המתודות שלה virtual ... = 0 (pure virtual) + destructor וירטואלי:

cpp · ICommand.h
#ifndef ICOMMAND_H
#define ICOMMAND_H
#include <string>
#include <vector>

// אינטרפייס לפקודות CLI. כל פקודה (add, recommend, help) ממומשת
// כמחלקה נפרדת — מאפשר להוסיף פקודות בלי לשנות קוד קיים.
class ICommand {
public:
    virtual ~ICommand() = default;
    // מריץ את הפקודה עם הארגומנטים (הטוקנים אחרי שם הפקודה)
    virtual std::string execute(const std::vector<std::string>& args) = 0;
    // מחזיר את מילת המפתח של הפקודה (e.g. "add")
    virtual std::string getName() const = 0;
};
#endif

1.2 פקודה בודדת XCommand.{h,cpp}

כל פקודה היא מחלקה שיורשת מ-ICommand, מקבלת את האחסון דרך הבנאי (Dependency Injection) ומחזיקה אותו כ-reference:

cpp · XCommand.h
#ifndef X_COMMAND_H
#define X_COMMAND_H
#include "ICommand.h"
#include "FileDataStorage.h"   // התלות שהפקודה צריכה

class XCommand : public ICommand {
private:
    FileDataStorage& storage;          // reference — הוזרק מבחוץ
public:
    XCommand(FileDataStorage& storage);                                  // בנאי
    std::string execute(const std::vector<std::string>& args) override;  // המתודות מהאינטרפייס
    std::string getName() const override;
};
#endif

#include "XCommand.h"

// initializer list — מאתחל את ה-reference

1.3 הפרסר CommandParser.{h,cpp}

הפרסר מנתב בלבד — הוא לא יודע מה כל פקודה עושה. מחזיק map משם-פקודה לאובייקט הפקודה. מקבל שורה, מפצל לפי רווחים: הטוקן הראשון = שם הפקודה, השאר = ארגומנטים.

cpp · CommandParser.cpp
#include "CommandParser.h"
#include <sstream>

void CommandParser::registerCommand(const std::string& name,
                                    std::shared_ptr<ICommand> command) {
    commands[name] = command;
}

void CommandParser::parse(const std::string& line) {
    std::istringstream s(line);
    std::string command;
    s >> command;                          // טוקן ראשון = שם הפקודה

    if (commands.find(command) == commands.end()) {
        return;                            // פקודה לא מוכרת — מתעלמים (לפי השאלה)
    }
    std::vector<std::string> args;
    std::string token;
    while (s >> token) {                    // שאר הטוקנים = ארגומנטים
        args.push_back(token);
    }
    commands[command]->execute(args);
}
🔤
std::istringstream עטיפה של מחרוזת ב-istringstream מאפשרת לשאוב ממנו מילה-מילה (לפי רווחים) עם האופרטור >> — בדיוק כמו קריאה מ-cin.

1.5 ה-main — תמיד אותו סדר

cpp · main.cpp
int main(int argc, char* argv[]) {
    // 1. יצירת האחסון (אם יש)
    FileDataStorage storage("data/users.txt");

    // 2. יצירת הפקודות (מזריקים להן את האחסון)
    auto postCmd = std::make_shared<PostCommand>(storage);
    auto getCmd  = std::make_shared<GetCommand>(storage);

    // 3. יצירת הפרסר ורישום הפקודות
    CommandParser parser;
    parser.registerCommand("POST", postCmd);
    parser.registerCommand("GET",  getCmd);

    // 4. לולאת הקלט — לא עוצרת לעולם
    std::string line;
    while (std::getline(std::cin, line)) {
        parser.parse(line);
    }
}
🔁
הזרימה המלאה משתמש מקליד → main קורא שורה → parser.parse(line) מפצל ומזהה → command.execute(args) מאמת ומבצע → storage כותב/קורא → מחזירים פלט (cout) → חזרה לאיטרציה הבאה.

2.1 TCP מול UDP — קריאות המערכת

צדקריאות לפי סדר
TCPשרתsocket → bind → listen → accept → recv/send → close
לקוחsocket → connect → send/recv → close
UDPשרתsocket → bind → recvfrom → sendto (ללא חיבור)
לקוחsocket → sendto → recvfrom → close
קריאהמה היא עושה
socket()יוצר את הסוקט (נקודת תקשורת)
bind()"תופס" את הפורט שהשרת יאזין לו
listen()מתחיל להאזין לחיבורים נכנסים
accept()חוסם וממתין ללקוח; כשמתחבר — מחזיר fd חדש לאותו לקוח

2.2 הלקוח

השלבים: (1) לקרוא ip+port מ-argv. (2) socket() + connect(). (3) לולאה אינסופית: קלט מהמשתמש → לשלוח לשרת → לקרוא תשובה → להדפיס.

python · client.py
import sys, socket

ip   = sys.argv[1]
port = int(sys.argv[2])

s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)   # TCP
s.connect((ip, port))

while True:
    line = input()                          # קלט מהמשתמש
    s.sendall((line + '\n').encode())       # שולח לשרת (עם \n)
    response = s.recv(4096).decode()        # מקבל תשובה
    print(response, end='')                 # מדפיס
cpp · client (אותו רעיון)
const char* ip = argv[1];
int port = std::stoi(argv[2]);

int sock = socket(AF_INET, SOCK_STREAM, 0);

struct sockaddr_in serverAddr;
serverAddr.sin_family = AF_INET;
serverAddr.sin_port = htons(port);
inet_pton(AF_INET, ip, &serverAddr.sin_addr);

    line += "\n";
    send(sock, line.c_str(), line.size(), 0);

close(sock);

2.3 השרת — ה"מראה" של הלקוח

השרת הוא אותה ליבת תרגיל 1, עטופה בלולאת רשת. נחליט שהפקודות מחזירות string (במקום להדפיס), והשרת שולח אותו ללקוח. הזרימה: קורא פורט → יוצר אחסון+פרסר+פקודות (כמו תרגיל 1) → socket → bind → listen → לולאת accept → קורא שורה בייט-בייט עד \n → parse → send תשובה.

cpp · server_main.cpp
int main(int argc, char* argv[]) {
    int port = std::stoi(argv[1]);

    // ---- ליבה (זהה ל-console) ----
    FileDataStorage storage("data/users.txt");
    CommandParser parser;
    parser.registerCommand("POST", std::make_shared<PostCommand>(storage));
    parser.registerCommand("GET",  std::make_shared<GetCommand>(storage));

    // ---- שכבת הרשת ----
    int serverFd = socket(AF_INET, SOCK_STREAM, 0);
    struct sockaddr_in addr;
    memset(&addr, 0, sizeof(addr));
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = INADDR_ANY;
    addr.sin_port = htons(port);
    bind(serverFd, (struct sockaddr*)&addr, sizeof(addr));
    listen(serverFd, 1);

    // ---- לולאת לקוחות ----
    while (true) {
        int clientFd = accept(serverFd, nullptr, nullptr);
        std::string line;
        char c;
        while (recv(clientFd, &c, 1, 0) > 0) {        // קורא בייט-בייט עד '\n'
            if (c == '\n') {
                std::string response = parser.parse(line);
                send(clientFd, response.c_str(), response.size(), 0);
                line.clear();
            } else {
                line += c;
            }
        }
        close(clientFd);
    }
    return 0;
}
⚠️
השינוי המהותי מתרגיל 1 ב-client-server, parse ו-execute מחזירים std::string (לא void ולא cout). השרת לוקח את ה-string ושולח אותו ב-send. כל השאר זהה.

  • Controller — המתווך. קורא בקשה ב-req, פונה ל-Model, עונה ב-res.

3.4 שלב 3 — Model (CRUD גנרי)

CRUD = הפעולות הבסיסיות על כל דאטה: Create (POST/add), Read (GET/getAll,findById), Update (PATCH/update), Delete (DELETE/delete). כדי שיהיה SOLID — כותבים Model.js גנרי אחד, וכל ישות יוצרת ממנו instance נפרד.

javascript · Model.js
class Model {
    #items = [];      // הרשימה בזיכרון (private)
    #nextId = 1;      // רץ אוטומטי

    getAll() { return [...this.#items]; }                       // R

    findById(id) { return this.#items.find(i => i.id === id); } // R (undefined אם אין)

    add(data) {                                                // C
        const item = { id: this.#nextId++, ...data };
        this.#items.push(item);
        return item;
    }
    update(id, data) {                                         // U
        const item = this.findById(id);
        if (!item) return null;
        Object.assign(item, data);
        return item;
    }
    delete(id) {                                               // D
        const index = this.#items.findIndex(i => i.id === id);
        if (index === -1) return false;
        this.#items.splice(index, 1);
        return true;
    }
}
module.exports = Model;
javascript · restaurants.js (instance לכל ישות)

module.exports = new Model();   // מאגר נפרד, רשימה ריקה משלו

דפוס התשובה — תמיד res.status(code).json(data) res.json(data) = 200 + JSON (ברירת מחדל). res.status(201).json(data) = קוד אחר. res.status(404).json({error:"..."}) = שגיאה. res.status(204).send() = בלי גוף. תמיד עונים פעם אחת בלבד — לכן משרשרים תנאים עם return.
javascript · restaurantController.js
const Restaurant = require('../models/restaurants');

function getAllRestaurants(req, res) {
    return res.json(Restaurant.getAll());                 // 200 + מערך (גם [])
}
function createRestaurant(req, res) {
    const data = req.body;
    if (!data.name) return res.status(400).json({ error: "Name is required" });   // 400
    const restaurant = Restaurant.add(data);
    return res.status(201).json(restaurant);              // 201
}
function getRestaurantById(req, res) {
    const id = parseInt(req.params.id);                   // params זה string!
    const restaurant = Restaurant.findById(id);
    if (!restaurant) return res.status(404).json({ error: "Not found" });          // 404
    return res.json(restaurant);                          // 200
}
function updateRestaurant(req, res) {
    const id = parseInt(req.params.id);
    if (!Restaurant.findById(id)) return res.status(404).json({ error: "Not found" });
    Restaurant.update(id, req.body);
    return res.status(204).send();                        // 204, גוף ריק
}
function deleteRestaurant(req, res) {
    const id = parseInt(req.params.id);
    if (!Restaurant.findById(id)) return res.status(404).json({ error: "Not found" });
    Restaurant.delete(id);
    return res.status(204).send();                        // 204
}
module.exports = { getAllRestaurants, createRestaurant, getRestaurantById,
                   updateRestaurant, deleteRestaurant };
🔄
מי קורא/כותב את req/res? — Express הלקוח שולח HTTP → Express מפענח ובונה req → מסתכל על METHOD+URL ומוצא את הנתיב → קורא לפונקציה שלנו עם (req,res) → הקונטרולר ממלא את res → Express הופך אותו ל-HTTP ושולח ללקוח.

3.7 שלב 6 — Middleware (לא תמיד)

פונקציה שרצה בין קבלת הבקשה לבין ה-Controller. מקבלת 3 פרמטרים — (req, res, next). next() = "סיימתי, תעביר הלאה". כל בקשה חייבת להסתיים או ב-next() או ב-res — לא בשתיהן.

javascript · auth middleware
function auth(req, res, next) {
    const userId = req.headers['authorization'];   // קורא את הזהות
    if (!userId) {
        return res.status(401).json({ error: "Unauthorized" });   // אין? → עצור
    }
    req.user = { userId };                          // שם על req ל-controllers
    next();                                          // המשך
}
module.exports = auth;

4.3 useState ואירועים

קומפוננטה היא פונקציה — היא רצה, מחזירה, ושוכחת. State הוא הזיכרון שלה, וכל פעם שהוא משתנה React מצייר מחדש אוטומטית.

jsx · useState + event
function Counter() {
    const [count, setCount] = React.useState(0);   // ערך + פונקציית שינוי
    return (
        <div>
            <p>You clicked {count} times</p>
            <button onClick={() => setCount(count + 1)}>Click me</button>
        </div>
    );
}
⚠️
לעולם לא לשנות state ישירות רק דרך הפונקציה (setCount) — כי היא לא רק מעדכנת את הערך, היא גם אומרת ל-React לצייר מחדש. אירועים נפוצים: onClick (לחיצה), onChange={(e) => setX(e.target.value)} (הקלדה). מעבירים את הפונקציה — לא מפעילים אותה עם ().

4.5 fetch + useEffect — מדברים עם השרת

fetch = בקשת HTTP מתוך JS (כמו curl), אסינכרונית. שני סוגים: להביא דאטה (GET) ולשלוח דאטה (POST).

javascript · fetch GET
const res  = await fetch('/api/restaurants');   // שולח בקשה
const data = await res.json();                  // הופך את התשובה ל-JSON
javascript · fetch POST
await fetch('/api/restaurants', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },   // שים לב — JSON
    body: JSON.stringify({ name, address })            // אובייקט → מחרוזת טקסט
});
📦
למה JSON.stringify? דרך הרשת אי אפשר לשלוח אובייקט JS — רק טקסט. הפעולה הופכת אובייקט למחרוזת. בצד השני (השרת) req.body ממיר חזרה ל-JSON.

איפה מציבים את ה-fetch? לא בגוף הקומפוננטה (ירוץ בכל ציור מחדש, אינסוף). משתמשים ב-useEffect כדי שירוץ פעם אחת בטעינה:

jsx · Home עם fetch + useEffect
function Home() {
    const [restaurants, setRestaurants] = React.useState([]);   // 1. state ריק

    React.useEffect(() => {                                      // 2. בטעינת המסך
        async function load() {
            const res  = await fetch('/api/restaurants');       // 3. הבא
            const data = await res.json();
            setRestaurants(data);                               // 4. שמור ב-state
        }
        load();
    }, []);                                                     // [] = פעם אחת בלבד

    return (
        <div>
            {restaurants.map(r => (                              // 5. הצג עם map
                <RestaurantCard key={r.id} restaurant={r} />
            ))}
        </div>
    );
}
🔁
סדר הריצה (1) גוף הפונקציה רץ. (2) ה-return מצייר JSX. (3) רק אחרי הציור רץ useEffect → fetch → setState. (4) שינוי ה-state → ציור מחדש, אבל הפעם useEffect לא ירוץ (כי []). אם נכניס [x] — ירוץ בכל פעם ש-x משתנה.

A תרגיל 2 — ספר טלפונים: שרת C++ (TCP) + לקוח Python

📝
השאלה שרת C++ על TCP המנהל ספר טלפונים, ולקוח Python. הפורט מועבר כארגומנט ל-main של השרת; ip+port ללקוח. לא עוצר לעולם. פקודות: ADD [name] [phone] (קיים → 409, אחרת → 201), GET [name] (לא קיים → 404, אחרת → 200 + הטלפון), DELETE [name] (לא קיים → 404, אחרת → 204), כל פקודה אחרת → 400. המידע נשמר לקובץ.
cpp · ContactStorage.cpp (in-memory map → קובץ)
#include "ContactStorage.h"
#include <fstream>
#include <sstream>

ContactStorage::ContactStorage(const std::string& filePath) : filePath(filePath) {
    std::ifstream file(filePath);
    std::string line;
    while (std::getline(file, line)) {       // טוען אנשי קשר קיימים
        std::istringstream s(line);
        std::string name, phone;
        s >> name >> phone;
        if (!name.empty()) contacts[name] = phone;
    }
}
bool ContactStorage::exists(const std::string& name) {
    return contacts.find(name) != contacts.end();
}
std::string ContactStorage::get(const std::string& name) { return contacts[name]; }

void ContactStorage::add(const std::string& name, const std::string& phone) {
    contacts[name] = phone;  save();
}
void ContactStorage::remove(const std::string& name) {
    contacts.erase(name);    save();
}
void ContactStorage::save() {                // דורס את כל הקובץ
    std::ofstream file(filePath);
    for (auto& pair : contacts)
        file << pair.first << " " << pair.second << "\n";
}
cpp · AddCommand / GetCommand / DeleteCommand (execute)
std::string AddCommand::execute(const std::vector<std::string>& args) {
    if (args.size() != 2) return "400 Bad Request\n";       // name + phone
    if (storage.exists(args[0])) return "409 Conflict\n";   // כבר קיים
    storage.add(args[0], args[1]);
    return "201 Created\n";
}
std::string GetCommand::execute(const std::vector<std::string>& args) {
    if (args.size() != 1) return "400 Bad Request\n";
    if (!storage.exists(args[0])) return "404 Not Found\n";
    return "200 Ok\n" + storage.get(args[0]) + "\n";        // סטטוס ואז הטלפון
}
std::string DeleteCommand::execute(const std::vector<std::string>& args) {
    if (args.size() != 1) return "400 Bad Request\n";
    if (!storage.exists(args[0])) return "404 Not Found\n";
    storage.remove(args[0]);
    return "204 No Content\n";
}

השאלה שרת Express בארכיטקטורת MVC, REST API לניהול ספרים, המידע בזיכרון. /api/books: GET → 200, POST (חסר title/author → 400, אחרת 201). /api/books/:id: GET/PATCH/DELETE (לא נמצא → 404; PATCH/DELETE מצליחים → 204).
javascript · src/controllers/bookController.js
const Book = require('../models/books');

function getAllBooks(req, res) {
    return res.json(Book.getAll());                                  // 200 + מערך
}
function createBook(req, res) {
    const data = req.body;
    if (!data.title || !data.author)
        return res.status(400).json({ error: "title and author are required" });
    return res.status(201).json(Book.add(data));                    // 201
}
function getBookById(req, res) {
    const book = Book.findById(parseInt(req.params.id));
    if (!book) return res.status(404).json({ error: "Book not found" });
    return res.json(book);                                          // 200
}
function updateBook(req, res) {
    const id = parseInt(req.params.id);
    if (!Book.findById(id)) return res.status(404).json({ error: "Book not found" });
    Book.update(id, req.body);
    return res.status(204).send();                                  // 204
}
function deleteBook(req, res) {
    const id = parseInt(req.params.id);
    if (!Book.findById(id)) return res.status(404).json({ error: "Book not found" });
    Book.delete(id);
    return res.status(204).send();                                  // 204
}
module.exports = { getAllBooks, createBook, getBookById, updateBook, deleteBook };

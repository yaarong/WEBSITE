Expo Router — list + add + delete

אותה אפליקציית אלבומים בשתי גרסאות: מקרה 1 — הכל במסך אחד; מקרה 2 — שני מסכים עם ניווט. כל הקוד, מוכן להעתקה.

🧭
מה שצריך לזכור על Expo Router File-based routing: כל קובץ ב-app/ = מסך. index.js = הבית (/), add.js = המסך /add. app/_layout.js עוטף הכל ב-<Stack /> (ניווט-ערימה). ניווט בקוד: const router = useRouter() → router.push('/add') (מעבר קדימה), router.back() (חזרה). BASE_URL עם 10.0.2.2 = localhost של המחשב מתוך האמולטור.

2. app/index.js — Home

js · app/index.js
import React, { useState, useEffect } from 'react';
import { View, TextInput, Button, FlatList, StyleSheet } from 'react-native';
import AlbumCard from '../components/AlbumCard';

const BASE_URL = 'http://10.0.2.2:3000';

export default function Home() {
  const [albums, setAlbums] = useState([]);
  const [title, setTitle]   = useState('');
  const [artist, setArtist] = useState('');
  const [year, setYear]     = useState('');

  async function loadAlbums() {
    const res = await fetch(`${BASE_URL}/api/albums`);
    setAlbums(await res.json());
  }

  useEffect(() => { loadAlbums(); }, []);

  async function handleAdd() {
    await fetch(`${BASE_URL}/api/albums`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ title, artist, year })
    });
    setTitle(''); setArtist(''); setYear('');
    loadAlbums();
  }

  async function handleDelete(id) {
    await fetch(`${BASE_URL}/api/albums/${id}`, { method: 'DELETE' });
    loadAlbums();
  }

  return (
    <View style={styles.container}>
      <TextInput style={styles.input} placeholder="Title"  value={title}  onChangeText={setTitle} />
      <TextInput style={styles.input} placeholder="Artist" value={artist} onChangeText={setArtist} />
      <TextInput style={styles.input} placeholder="Year"   value={year}   onChangeText={setYear} keyboardType="numeric" />
      <Button title="Add Album" onPress={handleAdd} />

      <FlatList
        data={albums}
        keyExtractor={item => item._id}
        renderItem={({ item }) => <AlbumCard album={item} onDelete={handleDelete} />}
      />
    </View>
  );
}

const styles = StyleSheet.create({
  container: { flex: 1, padding: 20 },
  input: { borderWidth: 1, padding: 8, marginBottom: 8 }
});

import React from 'react';
import { View, Text, Button, StyleSheet } from 'react-native';

export default function AlbumCard({ album, onDelete }) {
  return (
    <View style={styles.card}>
      <Text style={styles.title}>{album.title}</Text>
      <Text>{album.artist} · {album.year}</Text>
      <Button title="Delete" onPress={() => onDelete(album._id)} />

  );

const styles = StyleSheet.create({

  title: { fontWeight: 'bold', fontSize: 18 }
});

import React, { useState, useEffect } from 'react';
import { View, Button, FlatList, StyleSheet } from 'react-native';
import { useRouter } from 'expo-router';
import AlbumCard from '../components/AlbumCard';

const BASE_URL = 'http://10.0.2.2:3000';

export default function Home() {
  const router = useRouter();
  const [albums, setAlbums] = useState([]);

  async function loadAlbums() {
    const res = await fetch(`${BASE_URL}/api/albums`);
    setAlbums(await res.json());
  }

  useEffect(() => { loadAlbums(); }, []);

  async function handleDelete(id) {
    await fetch(`${BASE_URL}/api/albums/${id}`, { method: 'DELETE' });
    loadAlbums();
  }

  return (
    <View style={styles.container}>
      <Button title="Add Album" onPress={() => router.push('/add')} />
      <FlatList
        data={albums}
        keyExtractor={item => item._id}
        renderItem={({ item }) => <AlbumCard album={item} onDelete={handleDelete} />}
      />
    </View>
  );
}

const styles = StyleSheet.create({ container: { flex: 1, padding: 20 } });

3. app/add.js — טופס, ואחרי POST → router.back()

js · app/add.js
import React, { useState } from 'react';
import { View, TextInput, Button, StyleSheet } from 'react-native';
import { useRouter } from 'expo-router';

const BASE_URL = 'http://10.0.2.2:3000';

export default function Add() {
  const router = useRouter();
  const [title, setTitle]   = useState('');
  const [artist, setArtist] = useState('');
  const [year, setYear]     = useState('');

  async function handleAdd() {
    await fetch(`${BASE_URL}/api/albums`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ title, artist, year })
    });
    router.back();
  }

  return (
    <View style={styles.container}>
      <TextInput style={styles.input} placeholder="Title"  value={title}  onChangeText={setTitle} />
      <TextInput style={styles.input} placeholder="Artist" value={artist} onChangeText={setArtist} />
      <TextInput style={styles.input} placeholder="Year"   value={year}   onChangeText={setYear} keyboardType="numeric" />
      <Button title="Save" onPress={handleAdd} />
    </View>
  );
}

const styles = StyleSheet.create({
  container: { flex: 1, padding: 20 },
  input: { borderWidth: 1, padding: 8, marginBottom: 8 }
});

חלק א' · MongoDB Shell

1. התקנה ו-databases/collections

🧭
קונטקסט מתקינים את MongoDB Community + ה-shell mongosh. מסד מכיל databases, שמכילים collections (≈ טבלאות), שמכילים documents (≈ שורות, בפורמט JSON). זוכרים מהרצאה 9 — schema-less.
mongosh · ניהול DB ו-collections
show dbs                     // רשימת מסדים
use myapp                    // עובר/יוצר מסד בשם myapp
db.createCollection("users") // יוצר collection
show collections
db.users.find()              // כל ה-documents

// collection עם הגבלות (capped):
db.createCollection("games", { capped: true, size: 10000000, max: 100 })
db.games.drop()              // מחיקת collection

2. הכנסת נתונים

mongosh · insertOne / insertMany
db.users.insertOne({ name: "zvi", grade: 59 })

db.users.insertMany([
    { name: "berger", grade: 80 },
    { name: "zvika",  grade: 70 },
    { name: "db",     grade: 100 }
])

// document יכול להכיל טיפוסים מגוונים, מערכים ואובייקטים מקוננים:
db.users.insertOne({
    name: "Israel Unisraeli",
    age: 14,
    grade: 59.3,
    student: true,
    registerDate: new Date(),
    friends: null,
    courses: ["Advanced Programming", "Data Base", "Algorithm"],
    Adress: { street: "Israelia", city: "Mini Israel", zip: 123456 }
})
🔤
מבנה document document הוא אובייקט JSON גמיש — שדות יכולים להיות מספרים, מחרוזות, בוליאנים, תאריכים (new Date()), null, מערכים ואובייקטים מקוננים. אין צורך שכל ה-documents באותו collection יהיו זהים.

3. עדכון ומחיקה

mongosh · update / delete
db.users.updateOne(  { name: "db" }, { $set: { grade: 80 } } )    // עדכון שדה
db.users.updateMany( { },           { $set: { year: 2025 } } )   // כל ה-documents
db.users.updateOne(  { name: "db" }, { $unset: { year: "" } } )  // הסרת שדה
db.users.updateOne(  { name: "zvi"}, { $rename: { year: "date" } } )  // שינוי שם שדה

db.users.deleteOne({ name: "zvi" })
🔤
update operators הארגומנט הראשון = מי לעדכן (filter), השני = מה. $set (קובע ערך), $unset (מסיר שדה), $rename (משנה שם שדה). ...One = הראשון שמתאים, ...Many = כולם.

4. Find ו-query operators

mongosh · find + projections
db.users.find()                              // הכל
db.users.findOne({ grade: 80 })              // הראשון שמתאים
db.users.find({ grade: 80 })                 // כל מי ש-grade==80

// Projection — אילו שדות להחזיר (1/true=כן, 0/false=לא):
db.users.find({ grade: 80 }, { name: 1 })          // רק name (ו-_id)
db.users.find({ grade: 80 }, { _id: 0, name: 1 })  // רק name, בלי _id
mongosh · comparison + logical operators
db.users.find({ grade: { $gt: 70 } })        // greater than
db.users.find({ grade: { $gte: 70 } })       // greater or equal
db.users.find({ name: { $in: ["zvika", "berger"] } })   // ברשימה
db.users.find({ $and: [ { grade: { $gte: 70 } }, { year: { $ne: 2024 } } ] })

// מיון והגבלה:
db.users.find({}, { _id: 0 }).sort({ name: 1 })          // עולה
db.users.find({}, { _id: 0 }).sort({ grade: -1 }).limit(2)  // יורד, 2 ראשונים
🔤
אופרטורים נפוצים השוואה: $gt/$gte/$lt/$lte, $eq/$ne, $in (ברשימה). לוגיים: $and/$or. .sort({field: 1}) = עולה, -1 = יורד. .limit(n) = רק n ראשונים. כולם מתחילים ב-$ (אופרטור).

5. Aggregate ו-Indexing

mongosh · aggregate
// קיבוץ לפי שדה + חישוב — ממוצע ציון לכל שנה:
db.users.aggregate([ { $group: { _id: "$year", averageGrade: { $avg: "$grade" } } } ])
db.users.aggregate([ { $group: { _id: "$year", SumGrade: { $sum: "$grade" } } } ])
db.users.aggregate([ { $group: { _id: "$year", MaxGrade: { $max: "$grade" } } } ])
mongosh · indexing
db.users.find({ name: "berger" }).explain("executionStats")  // ניתוח ביצועים
db.users.createIndex({ name: 1 })    // יוצר אינדקס על name
db.users.getIndexes()
db.users.dropIndex("name_1")
🔤
Aggregate ו-Index
  • Aggregate — pipeline של שלבים. $group מקבץ לפי _id (השדה לקיבוץ, עם $) ומחשב צבירה ($avg/$sum/$min/$max).
  • Index — מבנה שמאיץ חיפושים על שדה (כמו אינדקס בספר). createIndex מאיץ find על אותו שדה משמעותית (במחיר זיכרון וכתיבה).

חלק ב' · שרת Express + Mongoose

6. הקמת השרת + Schema

bash · הקמת הפרויקט
mkdir nodejs-mongodb-api && cd nodejs-mongodb-api
npm init -y
npm install express mongoose cors dotenv
npm install nodemon --save-dev      # מריץ-מחדש אוטומטית בפיתוח
js · server.js — הקמה + Schema
const express = require('express');
const mongoose = require('mongoose');
const cors = require('cors');
require('dotenv').config();

const app = express();
app.use(cors());
app.use(express.json());      // מפענח גוף JSON ל-req.body

mongoose.connect(process.env.MONGODB_URI || 'mongodb://localhost:27017/myapp')
    .then(() => console.log('MongoDB connected successfully'))
    .catch(err => console.error('MongoDB connection error:', err));

const userSchema = new mongoose.Schema({
    name:  { type: String, required: true },
    email: { type: String, required: true, unique: true },
    age:   Number,
    createdAt: { type: Date, default: Date.now }
});
const User = mongoose.model('User', userSchema);
🔤
Schema של Mongoose מגדיר את מבנה ה-document עם טיפוסים ואילוצים: required (חובה), unique (ייחודי), default (ברירת מחדל). mongoose.model('User', schema) מחזיר את ה-Model שדרכו עושים את כל פעולות המסד (find, save...).

7. CRUD כ-REST API

🧭
קונטקסט כל פעולת CRUD ממופה למתודת HTTP ו-endpoint (REST — הרצאה 5/9). כל ה-handlers אסינכרוניים (async/await) ומחזירים JSON עם קוד סטטוס מתאים.
js · Read (GET)
// כל המשתמשים:
app.get('/api/users', async (req, res) => {
    try {
        const users = await User.find();
        res.json(users);
    } catch (err) {
        res.status(500).json({ message: err.message });
    }
});

// משתמש לפי id:
app.get('/api/users/:id', async (req, res) => {
    try {
        const user = await User.findById(req.params.id);
        if (!user) return res.status(404).json({ message: 'User not found' });
        res.json(user);
    } catch (err) {
        res.status(500).json({ message: err.message });
    }
});
js · Create (POST) + Update (PUT) + Delete
// יצירה:
app.post('/api/users', async (req, res) => {
    const user = new User({ name: req.body.name, email: req.body.email, age: req.body.age });
    try {
        const newUser = await user.save();
        res.status(201).json(newUser);          // 201 Created
    } catch (err) {
        res.status(400).json({ message: err.message });   // 400 Bad Request
    }
});

// עדכון:
app.put('/api/users/:id', async (req, res) => {
    const user = await User.findById(req.params.id);
    if (!user) return res.status(404).json({ message: 'User not found' });
    if (req.body.name) user.name = req.body.name;
    const updated = await user.save();
    res.json(updated);
});

// מחיקה:
app.delete('/api/users/:id', async (req, res) => {
    const user = await User.findById(req.params.id);
    if (!user) return res.status(404).json({ message: 'User not found' });
    await user.deleteOne();
    res.json({ message: 'User deleted successfully' });
});
CRUDHTTPEndpointסטטוס
CreatePOST/api/users201 Created
ReadGET/api/users · /api/users/:id200 / 404
UpdatePUT/api/users/:id200 / 404
DeleteDELETE/api/users/:id200 / 404
🔤
מתודות Mongoose ו-error handling User.find() (הכל), User.findById(id) (לפי id), new User({...}).save() (יצירה/עדכון), user.deleteOne() (מחיקה). עוטפים ב-try/catch ומחזירים קוד סטטוס מתאים (404 לא-נמצא, 400 קלט שגוי, 500 שגיאת שרת). בסוף אפשר גם middleware כללי לטיפול בשגיאות.

8. dotenv וקונפיגורציה

text · .env
MONGODB_URI=mongodb://localhost:27017/myapp
PORT=3000
js · שימוש
require('dotenv').config();              // טוען את .env ל-process.env

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => console.log(`Server is running on port ${PORT}`));
🔐
למה .env סודות (connection string, סיסמאות) וקונפיגורציה תלויית-סביבה לא נכתבים בקוד אלא בקובץ .env (שלא נכנס ל-git). dotenv טוען אותו ל-process.env. ה-|| 3000 נותן ערך ברירת מחדל אם המשתנה לא מוגדר.

2. let מול var — scoping

🔤
ההבדל המרכזי
  • let — משתנה ב-block scope (חי רק בתוך ה-{} שבו הוגדר). זו ברירת המחדל המומלצת.
  • var — משתנה ב-function scope (חי בכל הפונקציה), ניתן ל"שימוש" לפני הגדרה (כ-undefined), וניתן להגדיר אותו מחדש — מקור לבאגים.
js · block scope
// let — block scoped:
let x = 7;
{                       // פתיחת בלוק
    let x = 11;
    console.log(x);     // 11
}
console.log(x);         // 7  — ה-x החיצוני לא הושפע

// בלולאה: עם let, i לא קיים מחוץ ללולאה:
for (let i = 0; i < 3; i++) { console.log(`i = ${i}`); }
// console.log(i);      // ERROR — i לא מוגדר כאן

// var — function scoped, ואפשר להגדיר מחדש (לא מומלץ!):
var y = 2;
var y = 3;              // לא שגיאה עם var (עם let — שגיאה!)
console.log(y);         // 3

3. const — קבוע (אבל לא immutable)

🧭
נקודה עדינה וחשובה const מוגדר ב-block scope (כמו let), אבל לא ניתן לשנות את הערך שלו. ובכל זאת — אם זה מערך או אובייקט, לא ניתן להשים לתוכו ערך חדש, אבל כן ניתן לשנות את התוכן הפנימי!
js · const_values.js
const x = 3;
// x = 5;              // ❌ ERROR — לא ניתן לשנות ערך של const

const arr = [1, 2, 3];
// arr = [4, 5, 6];    // ❌ ERROR — לא ניתן להשים מערך חדש
arr[0] = 8;            // ✅ מותר! משנים תוכן: [8, 2, 3]

const car = { type: "Fiat", color: "white" };
car.color = "red";    // ✅ מותר! משנים שדה. אפילו אפשר להוסיף:
car.year = 1998;

4. פונקציות, template strings ו-setTimeout

דוגמה: ספירה לאחור (count down) — מדגימה פונקציות, מחרוזות-תבנית, ו-setTimeout:

js · count_down.js
function countDown(seconds) {
    const oneSecond = 1000;   // אלפיות שנייה
    console.log(`Time left ${seconds}`);

    for (let i = 1; i < seconds; i++) {
        // setTimeout(callback, delay, ...args) — מריץ את ה-callback אחרי delay
        setTimeout(updateClock, oneSecond * i, seconds - i);
    }

    setTimeout(() => {                       // arrow function כ-callback
        console.log("Time's Up!");
    }, oneSecond * seconds);
}

function updateClock(timeLeft) {
    console.log(`Time left: ${timeLeft}`);
}

countDown(5);
console.log("Some Stuff");   // יודפס *מיד* — לפני הספירה!
🔤
נקודות תחביר
  • Template string — מחרוזת ב-backticks `...` עם הזרקת ערכים ב-${...}.
  • setTimeout(fn, ms, ...args) — מתזמן הרצה של fn אחרי ms; אפשר להעביר ארגומנטים נוספים.
  • Arrow function — () => { ... }, תחביר קצר לפונקציה.
  • חשוב: setTimeout הוא אסינכרוני — הקוד לא מחכה לו! לכן "Some Stuff" מודפס לפני הספירה. (זוכרים אסינכרוניות מהרצאות ה-JS?)

5. אובייקט Date

js · current_date.js
const daylist = ["Sunday", "Monday", "Tuesday", "Wednesday",
                 "Thursday", "Friday", "Saturday"];

const today = new Date();        // אובייקט התאריך הנוכחי
const day = today.getDay();      // 0..6
console.log(`Today is: ${daylist[day]}.`);

let hour = today.getHours();     // 0..23
const minute = today.getMinutes();
const second = today.getSeconds();

// אופרטור טרנארי במקום if/else:
const period = (hour >= 12) ? "PM" : "AM";
hour = (hour >= 12) ? hour - 12 : hour;

console.log(`Current Time: ${hour} ${period} : ${minute} : ${second}`);
🔤
אופרטור טרנארי condition ? valueIfTrue : valueIfFalse — קיצור ל-if/else שמחזיר ערך. נוח להשמה מותנית.

6. רקורסיה ו-Memoization

js · fibonacci.js
// רקורסיה פשוטה — איטית (חישובים חוזרים), נתקעת על n גדול
function fibonacci(n) {
    if (n <= 1) return n;
    return fibonacci(n - 1) + fibonacci(n - 2);
}

// עם memoization — שומרים תוצאות שכבר חושבו במילון
function fibonacci_dm(n, memo = {}) {
    if (n in memo) return memo[n];
    if (n <= 1) return n;
    memo[n] = fibonacci_dm(n - 1, memo) + fibonacci_dm(n - 2, memo);
    return memo[n];
}

console.log(fibonacci_dm(300));   // מהיר!
⚡
Memoization טכניקת אופטימיזציה: שומרים תוצאות של קריאות שכבר חושבו (כאן ב-memo, אובייקט שמשמש כמילון), ובקריאה חוזרת מחזירים את הערך השמור במקום לחשב שוב. הופך את פיבונאצ'י מ-O(2ⁿ) ל-O(n). שימו לב לפרמטר עם ערך ברירת מחדל: memo = {}.

7. מערכים — forEach, filter, map

js · fun_with_arrays.js
// ספירת מופעים עם forEach:
function countItemOccurrence(array, item) {
    let counter = 0;
    array.forEach(x => { if (x == item) counter++; });
    return counter;
}

// סינון — מימוש ידני (קיים גם array.filter המובנה):
function filterArray(array, filterFunction) {
    let result = [];
    array.forEach(x => { if (filterFunction(x) === true) result.push(x); });
    return result;
}

// הכפלת כל איבר:
function multiplyEachElement(array, multiplier) {
    let newArray = [];
    array.forEach(x => newArray.push(x * multiplier));
    return newArray;
}

console.log(filterArray([1, 2, 3, '3'], x => x == '3'));     // [3, '3']
console.log(multiplyEachElement([1, 2, '3'], 5));            // [5, 10, 15]
🔤
מתודות מערך + higher-order functions
  • arr.forEach(fn) — מריץ fn על כל איבר (לא מחזיר מערך). שימו לב: אי אפשר break בתוך forEach — לכך משתמשים ב-for רגיל.
  • arr.push(x) — מוסיף לסוף. arr.length — אורך.
  • המובנים: arr.filter(fn), arr.map(fn) — מקבלים פונקציה כארגומנט (higher-order function) ומחזירים מערך חדש. עדיף עליהם על מימוש ידני.

8. אובייקטים ומחלקות

js · object literal + methods
// object literal:
const myObj = { attribute1: 'value1', attribute2: 'value2' };
console.log(myObj.attribute1);       // גישה עם נקודה
console.log(myObj['attribute2']);    // או עם סוגריים

// אובייקט עם מתודה — this מפנה לאובייקט:
const person = {
    firstName: "Israel",
    lastName: "Israeli",
    fullName: function() {
        return this.firstName + " " + this.lastName;
    }
};
console.log(person.fullName());      // "Israel Israeli"
js · class (עם שדות פרטיים #)
class Car {
    #fuel = 0;                       // שדה פרטי (קידומת #)

    constructor(name, capacity) {
        this.name = name;            // שדה ציבורי
        this.capacity = capacity;
    }

    fuelCar(amount) {
        if (this.#fuel + amount <= this.capacity) this.#fuel += amount;
    }
    currentFuel() { return this.#fuel; }
}

const car = new Car("Mazda", 500);
car.fuelCar(20);
console.log(car.currentFuel());      // 20

// אפשר להוסיף שדות/מתודות בזמן ריצה:
car.color = "blue";
🧊
Object.freeze — אובייקט immutable Object.freeze(obj) "מקפיא" אובייקט כך שלא ניתן לשנותו. ניסיון לשנות יזרוק שגיאה ב-strict mode (ובלעדיו פשוט לא יעשה כלום).

9. מלכודת — השוואת ==

🪤
type coercion האופרטור == מבצע המרת טיפוסים אוטומטית לפני ההשוואה — מה שמוביל לתוצאות מוזרות:
js
[] == 0        // true  (!)
"\t" == 0      // true  (!)
"0" == 0       // true
[] == "0"      // false (!)
כלל זהב: כמעט תמיד להשתמש ב-=== (השוואה קפדנית, בלי המרת טיפוסים) במקום ==. כך נמנעים מהפתעות.

★ סיכום

🎯
מה לקחנו
  • משתנים: let (block scope, מומלץ), var (function scope, להימנע), const (קבוע — אך תוכן מערך/אובייקט ניתן לשינוי).
  • פונקציות: רגילות ו-arrow (=>), template strings (`${}`), setTimeout אסינכרוני, אופרטור טרנארי.
  • מערכים: forEach/filter/map/push — higher-order functions.
  • אובייקטים: literals, מתודות + this, class עם שדות פרטיים (#), Object.freeze.
  • מלכודת: == ממיר טיפוסים — להשתמש ב-===.
